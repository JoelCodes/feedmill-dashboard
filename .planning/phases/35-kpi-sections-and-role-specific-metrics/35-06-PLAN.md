          return (
            <Card>
              <div className="p-4">
                <p className="text-sm font-bold text-[var(--text-primary)]">7-Day Volume Trend</p>
                <section role="img" aria-label="7-day production volume trend bar chart">

                    {data.map((d, i) => {
                      const barHeight = d.completedLbs === 0 ? 0 : Math.max(4, (d.completedLbs / maxLbs) * 120);
                      const x = i * 56;
                      const isToday = i === todayIdx;
                      return (
                        <g key={d.date}>
                          <rect x={x} y={120 - barHeight} width={48} height={barHeight}
                                fill="var(--primary)" opacity={isToday ? 1 : 0.8} />
                          <text x={x + 24} y={148} textAnchor="middle"
                                fontSize="11" fill="var(--text-muted)">

                        </g>

                    })}
                  </svg>
                </section>
              </div>
            </Card>
          );
        }
