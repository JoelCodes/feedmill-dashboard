<feature>
  <name>Customer service with stats aggregation</name>
  <files>src/services/customers.ts, src/services/customers.test.ts</files>
  <behavior>
    getCustomers() behavior:
    - Returns Promise<CustomerWithStats[]>
    - Each customer has stats computed from mockOrders and mockBins
    - stats.totalOrders = count of orders where order.customerId === customer.id
    - stats.activeOrders = count of orders where status !== "Complete"
    - stats.completedOrders = count of orders where status === "Complete"
    - stats.hasChanges = true if any customer order has hasChanges === true
    - stats.binAlertLevel = highest alert level among customer bins (critical > low > none)

    getCustomerById(id: string) behavior:
    - Returns Promise<CustomerWithStats | null>
    - Returns null if no customer with matching id exists
    - Returns customer with computed stats if found

    Test cases:
    - getCustomers() returns array with length 18 (all customers)
    - First customer has id property
    - First customer has stats object with totalOrders property
    - getCustomerById("CUST-001") returns customer with id "CUST-001"
    - getCustomerById("INVALID") returns null
    - Customer with orders has totalOrders > 0
    - Customer with completed orders has completedOrders > 0
    - Customer with bin at critical level has binAlertLevel "critical"
  </behavior>
