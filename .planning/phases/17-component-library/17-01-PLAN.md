1. Create test file first at `src/components/ui/Button.test.tsx`:
   - Import render, screen from @testing-library/react
   - Import Button from ./Button
   - Write failing tests for all 11 behaviors listed above

2. Create Button component at `src/components/ui/Button.tsx`:
   - Import cva, type VariantProps from "class-variance-authority"
   - Import cn from "@/lib/utils"
   - Import Loader2 from "lucide-react" for loading spinner

3. Define buttonVariants using cva():
   - Base classes: "inline-flex items-center justify-center rounded-[var(--radius-md)] font-medium transition-colors focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-offset-2 disabled:pointer-events-none disabled:opacity-50"
   - variant.primary: "bg-[var(--primary)] text-[var(--text-white)] hover:bg-[var(--primary-hover)] active:bg-[var(--primary-active)] disabled:bg-[var(--primary-disabled)] focus-visible:ring-[var(--primary)]"
   - variant.secondary: "bg-[var(--bg-card)] text-[var(--text-primary)] border border-[var(--divider)] hover:opacity-90 active:opacity-80"
   - variant.ghost: "bg-transparent text-[var(--text-primary)] hover:bg-[var(--bg-card)] active:opacity-80"
   - variant.destructive: "bg-[var(--error)] text-[var(--text-white)] hover:bg-[var(--error-hover)] active:bg-[var(--error-active)] disabled:bg-[var(--error-disabled)] focus-visible:ring-[var(--error)]"
   - size.sm: "h-8 px-3 text-sm gap-2"
   - size.md: "h-10 px-4 text-base gap-2"
   - size.lg: "h-12 px-6 text-lg gap-3"
   - defaultVariants: { variant: "primary", size: "md" }

4. Define ButtonProps interface:
   - Extends React.ButtonHTMLAttributes<HTMLButtonElement>
   - Extends VariantProps<typeof buttonVariants>
   - loading?: boolean
   - icon?: React.ReactNode

5. Implement Button function component:
   - Destructure variant, size, loading, icon, disabled, className, children, ...props
   - Return button element with:
     - className={cn(buttonVariants({ variant, size }), className)}
     - disabled={disabled || loading}
     - aria-disabled={disabled || loading || undefined}
     - aria-busy={loading || undefined}
     - {loading ? <Loader2 className="h-4 w-4 animate-spin" /> : icon}
     - {children}

6. Export: export default Button
